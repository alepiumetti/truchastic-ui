<script>
function togglemodal(action) {
if (action === 'open') {
let modal = document.getElementById('modal'); modal.setAttribute('show', 'true');
} else if (action === 'close') {
let modal = document.getElementById('modal'); modal.removeAttribute('show', 'false');
}
</script>
